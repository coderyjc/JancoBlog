垂直布局管理

  1. """

  3. """
  5. class Demo(QWidget):
  7.     def __init__(self):
  8.         super(Demo(),  self)

  12.         self.v_layout = QVBoxLayout()  # 实例化一个垂直布局管理器

水平垂直混合使用

  1. """
  2. 混合使用水平bu'ju
  3. """
  5. class Demo(QWidget):
  6.     def __init__(self):
  7.         super(Demo, self).__init__()
  9.         # 初始化按钮、文本行、标签等
  10.         self.user_label = QLabel('Username', self)
  11.         self.pwd_label = QLabel('Password', self)
  12.         self.user_line = QLineEdit(self)
  13.         self.pwd_line = QLineEdit(self)
  14.         self.login_button = QPushButton('log in', self)
  15.         self.signin_button = QPushButton('Sign in', self)
  17.         self.label_v_layout = QVBoxLayout()  # 用来管理label
  18.         self.line_v_layout = QVBoxLayout()  # 用来管理line
  19.         self.button_h_layout = QHBoxLayout()  # 用来管理button
  21.         # 将self.label_v_layout垂直布局和self.line_vlayout垂直布局这两个布局管理器从左到右依次水平摆放
  22.         self.label_line_h_layout = QHBoxLayout()
  23.         # 将self.label_line_h_layout和self.button_h_layout垂直从上到下摆放
  24.         self.all_v_layout = QVBoxLayout()
  25.         # 以上两个布局管理器用来管理1 - 3中的布局，我们应该通过addLayout()向其中添加布局管理器
  27.         self.label_v_layout.addWidget(self.user_label)
  28.         self.label_v_layout.addWidget(self.pwd_label)
  29.         self.line_v_layout.addWidget(self.user_line)
  31.         self.line_v_layout.addWidget(self.pwd_line)
  32.         self.button_h_layout.addWidget(self.login_button)
  33.         self.button_h_layout.addWidget(self.signin_button)
  35.         self.label_line_h_layout.addLayout(self.label_v_layout)
  36.         self.label_line_h_layout.addLayout(self.line_v_layout)
  37.         self.all_v_layout.addLayout(self.label_line_h_layout)
  38.         self.all_v_layout.addLayout(self.button_h_layout)
  39.         # 添加控件用addWidght()，添加布局用addLayout()
  41.         self.setLayout(self.all_v_layout)
  43. # 上面的代码是将两个QLabel用垂直布局方式摆放，将两个QLineEdit也用垂直布局方式摆放，最后用一个水平布局管理来摆放着两个垂直布局管理器。那换种思路，可以把QLabel和QLineEdit用水平布局方式摆放

表单布局

  1. """
  2. 表单布局 QFormLayout
  3. """
  5. # 表单布局可以将控件以两列的形式进行排布，左列控件为文本标签，右列为输入型的控件，如QLineEdit。
  7. class Demo(QWidget):
  9.     def __init__(self):
  10.         super(Demo, self).__init__()
  12.         self.user_label = QLabel('Username:', self)
  13.         self.pwd_label = QLabel('Password:', self)
  14.         self.user_line = QLineEdit(self)
  15.         self.pwd_line = QLineEdit(self)
  16.         self.login_button = QPushButton('Log in', self)
  17.         self.signin_button = QPushButton('Sign in', self)
  19.         self.f_layout = QFormLayout()  # 实例化一个QFormLayout控件
  21.         self.button_h_layout = QHBoxLayout()
  22.         self.all_v_layout = QVBoxLayout()
  24.         # 调用addRow()方法传入QLabel和QLineEdit控件
  25.         self.f_layout.addRow(self.user_label, self.user_line)  
  26.         self.f_layout.addRow(self.pwd_label, self.pwd_line)
  27.         self.button_h_layout.addWidget(self.login_button)
  28.         self.button_h_layout.addWidget(self.signin_button)
  30.         self.all_v_layout.addLayout(self.f_layout)  # 将表单布局添加到总布局中
  31.         self.all_v_layout.addLayout(self.button_h_layout)
  33.         self.setLayout()

网格布局

  1. """
  2. 网格布局 QGridLayout
  3. """
  5. # 相当于坐标式布局
  7. class Demo(QWidget):
  9.     def __init__(self):
  10.         super(Demo, self).__init__()
  12.         self.user_label = QLabel('Username:', self)
  13.         self.pwd_label = QLabel('Password:', self)
  14.         self.user_line = QLineEdit(self)
  15.         self.pwd_line = QLineEdit(self)
  16.         self.login_button = QPushButton('Log in', self)
  17.         self.signin_button = QPushButton('Sign in', self)
  19.         self.grid_layout = QGridLayout()  # 实例化一个QGridLayout布局管理器
  20.         self.h_layout = QHBoxLayout()
  21.         self.v_layout = QVBoxLayout()
  23.         self.grid_layout.addWidget(self.user_label, 0, 0, 1, 1)
  24.         self.grid_layout.addWidget(self.user_line, 0, 1, 1, 1)
  25.         self.grid_layout.addWidget(self.pwd_label, 1, 0, 1, 1)
  26.         self.grid_layout.addWidget(self.pwd_line, 1, 1, 1, 1)
  27.         # QGridLayout的addWidget()方法遵循如下语法形式：
  28.         # addWidget(widget, row, column, rowSpan, columnSpan)
  29.         # 在第几行，在第几列， 占几行， 占几列
  31.         self.h_layout.addWidget(self.login_button)
  32.         self.h_layout.addWidget(self.signin_button)
  34.         self.v_layout.addWidget(self.grid_layout)
  35.         self.v_layout.addWidget(self.h_layout)
  36.         # 最后，程序用垂直布局管理器将一个网格布局和一个水平布局添加进去。
  38.         self.setLayout(self.v_layout)
